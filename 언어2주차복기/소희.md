function solution(n) // Number 123
{
    let answer = 0;
    let my = String(n); // String "123"
    for ( let i=0; i < my.length; i++){
        // answer += Number(my[i]);  // 인덱스접근 // Number 1 / 2 / 3 
        answer += (+my[i]); // 2.형변환 방법 //3.parseInt()
    }
    return answer;
}
